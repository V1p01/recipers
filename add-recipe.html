document.addEventListener('DOMContentLoaded', function() { const form = document.getElementById('recipeForm'); const submitBtn = document.getElementById('submitRecipeBtn'); const successMsg = document.getElementById('successMessage'); const errorMsg = document.getElementById('errorMessage'); // Добавляем поле для капчи перед кнопкой const captchaDiv = document.createElement('div'); captchaDiv.className = 'form-group'; captchaDiv.innerHTML = ` `; submitBtn.parentNode.insertBefore(captchaDiv, submitBtn); if (form) { form.addEventListener('submit', async function(e) { e.preventDefault(); // Проверка капчи const captcha = document.getElementById('captchaAnswer'); const captchaHidden = document.getElementById('captchaHidden'); if (captcha.value.trim() !== '5') { alert('❌ Неправильный ответ! Попробуйте ещё раз.'); captcha.value = ''; captcha.style.borderColor = '#FF6B6B'; captcha.focus(); return; } captchaHidden.value = 'Пройдена'; submitBtn.textContent = 'Отправка...'; submitBtn.style.opacity = '0.7'; successMsg.classList.add('hidden'); errorMsg.classList.add('hidden'); try { const formData = new FormData(form); const response = await fetch(form.action, { method: 'POST', body: formData, headers: { 'Accept': 'application/json' } }); if (response.ok) { successMsg.classList.remove('hidden'); form.reset(); window.scrollTo({ top: 0, behavior: 'smooth' }); setTimeout(function() { successMsg.classList.add('hidden'); }, 5000); } else { throw new Error('Ошибка отправки'); } } catch (error) { console.error('Ошибка:', error); errorMsg.classList.remove('hidden'); setTimeout(function() { errorMsg.classList.add('hidden'); }, 5000); } submitBtn.textContent = 'Отправить рецепт'; submitBtn.style.opacity = '1'; }); } const requiredInputs = form.querySelectorAll('[required]'); requiredInputs.forEach(function(input) { input.addEventListener('invalid', function() { this.style.borderColor = '#FF6B6B'; }); input.addEventListener('input', function() { if (this.validity.valid) { this.style.borderColor = '#4ECDC4'; } }); }); });
